  <action>
    Create `app/api/test-mediainfo/route.js` exporting a `GET` handler. The handler:
    1. Accepts an optional `?url=` query param. Default: `"https://test-videos.co.uk/vids/bigbuckbunny/mp4/h264/360/Big_Buck_Bunny_360_10s_1MB.mp4"` (small public H.264 sample; if this URL becomes unreachable, executor may substitute any public sub-2MB MP4 URL).
    2. Fetches the URL with `fetch(url)`. If response not ok, return `Response.json({ ok: false, error: "fetch failed", status: res.status }, { status: 502 })`.
    3. Reads the body as ArrayBuffer, converts to Uint8Array.
    4. Calls `mediaInfoFactory({ format: 'object' })`, runs `analyzeData` exactly per the RESEARCH.md Pattern 3 example, wraps in try/finally with `mediainfo.close()`.
    5. Returns `Response.json({ ok: true, url, durationMs, trackCount: result?.media?.track?.length || 0, videoTrack: result?.media?.track?.find(t => t['@type'] === 'Video') || null })`.
    6. On any throw, return `Response.json({ ok: false, error: err.message }, { status: 500 })`.
    7. Set `export const runtime = 'nodejs'` (mediainfo.js WASM needs the Node runtime — NOT edge).
    8. Set `export const maxDuration = 30` so Vercel doesn't kill it during cold start + fetch + analyze.

    Full file:
    ```js
    import mediaInfoFactory from 'mediainfo.js';

    export const runtime = 'nodejs';
    export const maxDuration = 30;

    const DEFAULT_TEST_URL = 'https://test-videos.co.uk/vids/bigbuckbunny/mp4/h264/360/Big_Buck_Bunny_360_10s_1MB.mp4';

    export async function GET(request) {
      const { searchParams } = new URL(request.url);
      const url = searchParams.get('url') || DEFAULT_TEST_URL;
      const t0 = Date.now();
      let mediainfo = null;
      try {
        const videoRes = await fetch(url);
        if (!videoRes.ok) {
          return Response.json({ ok: false, error: 'fetch failed', status: videoRes.status, url }, { status: 502 });
        }
        const buffer = await videoRes.arrayBuffer();
        const uint8 = new Uint8Array(buffer);
        mediainfo = await mediaInfoFactory({ format: 'object' });
        const result = await mediainfo.analyzeData(
          () => uint8.byteLength,
          (chunkSize, offset) => uint8.slice(offset, offset + chunkSize)
        );
        const videoTrack = result?.media?.track?.find((t) => t['@type'] === 'Video') || null;
        return Response.json({
          ok: true,
          url,
          durationMs: Date.now() - t0,
          trackCount: result?.media?.track?.length || 0,
          videoTrack
        });
      } catch (err) {
        return Response.json({ ok: false, error: err?.message || String(err), durationMs: Date.now() - t0, url }, { status: 500 });
      } finally {
        try { mediainfo?.close?.(); } catch {}
      }
    }
    ```

    NOTE: This endpoint is intentionally PUBLIC — no admin auth — so it can be hit directly from `curl` on the deployed Vercel URL without cookies. It returns no sensitive data.
